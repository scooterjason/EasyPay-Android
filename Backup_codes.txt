		if("LTR".indexOf("RTL") == -1){

				var thisId = $(this).attr("id"); 
				//console.log(thisClassName);
				var html = $('<'+tag+'/>').attr('id',thisId).addClass(thisClassName).html(contents);
				console.log(html);

			});
